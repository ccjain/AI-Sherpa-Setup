# Phase 2b Implementation Plan — LLM-Assisted Analyzer

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add one optional LLM polish step inside the Phase 2a analyzer pipeline that produces PR-ready rule wording + unified diff for each new finding, attached as nullable columns on the existing `findings` table. Default OFF; opt-in via `Config.llm_enabled=True`.

**Architecture:** Single new module `server/analyzer/llm_draft.py` exporting `draft_polish(finding, cfg, budget)`. Schema migration adds 9 nullable columns to `findings` and 3 to `runs`. Pipeline calls `draft_polish` once per new finding (skipping recurrences and `confidence=="low"`); on any failure, the finding ships without polish. Renderer and auto-filer prefer LLM-drafted fields when present, fall back to Phase 2a behavior otherwise. Hard cost cap via `Budget` class (calls + tokens per run).

**Tech Stack:** Python 3.11+, `anthropic` SDK (≥0.30), `keyring` (Windows Credential Manager). Phase 2a server package required.

**Spec:** `docs/superpowers/specs/2026-05-29-phase2b-design.md` (commit `c44801f`).
**Builds on:** Phase 2a server package at `server/` (commit `f393b18`).

```python
"""Apply schemas to events.sqlite and insights.sqlite. Idempotent.

Phase 2b adds LLM-polish columns via ALTER. ALTERs are not idempotent in
SQLite; we catch and skip when the column already exists.
"""
from __future__ import annotations
import sqlite3
from server.config import Config
from server.storage.schema import EVENTS_DDL, INSIGHTS_DDL, LLM_MIGRATION_DDL


def migrate_all(cfg: Config) -> None:
    cfg.root.mkdir(parents=True, exist_ok=True)
    _apply(cfg.events_db, EVENTS_DDL)
    _apply(cfg.insights_db, INSIGHTS_DDL)
    _apply_alter(cfg.insights_db, LLM_MIGRATION_DDL)


def _apply(db_path, ddl: str) -> None:
    with sqlite3.connect(db_path) as conn:
        conn.executescript(ddl)
        conn.commit()


def _apply_alter(db_path, ddl: str) -> None:
    """Apply ALTER statements one at a time, skipping any that fail
    because the column already exists."""
    with sqlite3.connect(db_path) as conn:
        for stmt in [s.strip() for s in ddl.split(";") if s.strip() and not s.strip().startswith("--")]:
            try:
                conn.execute(stmt)
            except sqlite3.OperationalError as exc:
                if "duplicate column name" not in str(exc):
                    raise
        conn.commit()
```

Create `server/analyzer/llm_draft.py` (first half — Task 4 and 5 append):

```python
"""LLM-assisted polish for analyzer findings (Phase 2b).

Public surface:
- LlmDraft: dataclass holding LLM output
- Budget:   tracks per-run cost ceiling
- draft_polish(finding, cfg, budget) -> LlmDraft | None: main entry point

Internal:
- sanitize_finding_for_llm: strip PII before sending
- _scrub_text: regex-based PII stripper
- _validate_response: post-call JSON structure check
"""
from __future__ import annotations
import datetime
import json
import re
from dataclasses import dataclass
from server.config import Config
from server.analyzer.detectors.base import Finding


# ── Data shapes ────────────────────────────────────────────────────────

@dataclass(frozen=True)
class LlmDraft:
    refined_title: str
    target_file: str
    rule_wording: str
    unified_diff: str
    rationale: str
    model: str
    input_tokens: int
    output_tokens: int
    drafted_at: str


class Budget:
    """Per-run LLM cost ceiling. Methods are not thread-safe; pipeline runs sequentially."""
    def __init__(self, cfg: Config):
        self.calls_remaining = cfg.llm_max_calls_per_run
        self.input_tokens_remaining = cfg.llm_max_input_tokens_per_run
        self.calls_made = 0
        self.input_tokens_used = 0
        self.output_tokens_used = 0

    def can_call(self, est_input_tokens: int) -> bool:
        return (self.calls_remaining > 0 and
                self.input_tokens_remaining >= est_input_tokens)

    def record(self, input_tokens: int, output_tokens: int) -> None:
        self.calls_remaining -= 1
        self.input_tokens_remaining -= input_tokens
        self.calls_made += 1
        self.input_tokens_used += input_tokens
        self.output_tokens_used += output_tokens


# ── Sanitization ───────────────────────────────────────────────────────

_EMAIL_RE = re.compile(r"\b[\w.+-]+@[\w-]+\.[\w.-]+\b")
_UUID_RE = re.compile(r"\b[a-f0-9]{8}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{12}\b")
_WIN_PATH_RE = re.compile(r"[A-Z]:\\[^\s]+")
_UNIX_HOME_RE = re.compile(r"/(?:home|Users|tmp)/[^\s]+")


def _scrub_text(text: str) -> str:
    if not text:
        return ""
    text = _EMAIL_RE.sub("<email>", text)
    text = _UUID_RE.sub("<session_id>", text)
    text = _WIN_PATH_RE.sub("<path>", text)
    text = _UNIX_HOME_RE.sub("<path>", text)
    return text


def sanitize_finding_for_llm(finding: Finding) -> dict:
    """Strip developer names, file paths, session IDs from a finding."""
    return {
        "scenario_id": finding.scenario_id,
        "bucket": finding.bucket,
        "severity": finding.severity,
        "confidence": finding.confidence,
        "title": finding.title,
        "evidence_md": _scrub_text(finding.evidence_md),
        "sample_snippets_json": json.dumps([
            {"path": "<session_N>"} for _ in finding.sample_session_paths[:5]
        ]),
    }


# ── Response validation ────────────────────────────────────────────────

_TARGET_FILE_PATTERN = re.compile(
    r"^(core/CLAUDE\.md|domains/[a-z]+/CLAUDE\.md|skills/[a-z0-9-]+/SKILL\.md|"
    r"plugins\.json|setup\.(bat|sh|ps1)|docs/[a-z0-9-]+\.md)$"
)


def _validate_response(parsed: dict) -> bool:
    required = ("refined_title", "target_file", "rule_wording", "unified_diff", "rationale")
    if not all(k in parsed and isinstance(parsed[k], str) and parsed[k].strip() for k in required):
        return False
    if not _TARGET_FILE_PATTERN.match(parsed["target_file"]):
        return False
    if len(parsed["refined_title"]) > 70:
        return False
    if not (parsed["unified_diff"].startswith("--- ") or parsed["unified_diff"].startswith("diff --git")):
        return False
    return True
```

def draft_polish(finding: Finding, cfg: Config, budget: Budget) -> LlmDraft | None:
    """Call the configured LLM to draft PR-ready polish. Returns None on any failure."""
    if not cfg.llm_enabled:
        return None
    if not budget.can_call(est_input_tokens=2000):
        return None
    if cfg.llm_provider != "anthropic":
        # Path 2 (local LLM) not implemented in v1
        return None
    try:
        return _call_anthropic(finding, cfg, budget)
    except Exception:
        return None


def _call_anthropic(finding: Finding, cfg: Config, budget: Budget) -> LlmDraft | None:
    if keyring is None:
        return None
    api_key = keyring.get_password(cfg.llm_api_credential_name, "default")
    if not api_key:
        return None
    if Anthropic is None:
        return None

    sanitized = sanitize_finding_for_llm(finding)
    user_prompt = (
        f"Scenario: {sanitized['scenario_id']}\n"
        f"Bucket: {sanitized['bucket']}\n"
        f"Severity: {sanitized['severity']}\n"
        f"Confidence: {sanitized['confidence']}\n"
        f"Title: {sanitized['title']}\n\n"
        f"Evidence (Markdown):\n{sanitized['evidence_md']}\n\n"
        f"Sample snippets (sanitized):\n{sanitized['sample_snippets_json']}\n\n"
        f"Produce the JSON object as specified."
    )

    client = Anthropic(api_key=api_key)
    resp = client.messages.create(
        model=cfg.llm_model,
        max_tokens=2000,
        system=[{
            "type": "text",
            "text": _SYSTEM_PROMPT,
            "cache_control": {"type": "ephemeral"},
        }],
        messages=[{"role": "user", "content": user_prompt}],
    )

    body = resp.content[0].text if resp.content else ""
    try:
        parsed = json.loads(body)
    except json.JSONDecodeError:
        return None

    if not _validate_response(parsed):
        return None

    input_tokens = getattr(resp.usage, "input_tokens", 0)
    output_tokens = getattr(resp.usage, "output_tokens", 0)
    budget.record(input_tokens=input_tokens, output_tokens=output_tokens)

    return LlmDraft(
        refined_title=parsed["refined_title"],
        target_file=parsed["target_file"],
        rule_wording=parsed["rule_wording"],
        unified_diff=parsed["unified_diff"],
        rationale=parsed["rationale"],
        model=cfg.llm_model,
        input_tokens=input_tokens,
        output_tokens=output_tokens,
        drafted_at=datetime.datetime.utcnow().isoformat(),
    )

```python
"""Cost reporter — sums LLM token usage from runs.llm_* columns and
prints an estimated dollar figure.

Pricing assumptions (Haiku, current Anthropic pricing — confirm at
implementation time; defaults are conservative).
"""
from __future__ import annotations
import argparse
import datetime
import sqlite3
from server.config import Config


# Per-million token prices for Haiku (as of 2026-05). Override if needed.
PRICE_INPUT_PER_M_USD = 0.25
PRICE_OUTPUT_PER_M_USD = 1.25


def compute_cost(cfg: Config, window_days: int = 30) -> dict:
    cutoff = (datetime.datetime.utcnow() - datetime.timedelta(days=window_days)).isoformat()
    with sqlite3.connect(cfg.insights_db) as conn:
        row = conn.execute(
            "SELECT COALESCE(SUM(llm_input_tokens_total), 0), "
            "COALESCE(SUM(llm_output_tokens_total), 0), "
            "COALESCE(SUM(llm_calls_total), 0) "
            "FROM runs WHERE started_at >= ?",
            (cutoff,),
        ).fetchone()
    input_t, output_t, calls = row[0], row[1], row[2]
    cost = (input_t / 1_000_000) * PRICE_INPUT_PER_M_USD + (output_t / 1_000_000) * PRICE_OUTPUT_PER_M_USD
    return {
        "window_days": window_days,
        "calls": calls,
        "input_tokens": input_t,
        "output_tokens": output_t,
        "estimated_cost_usd": round(cost, 4),
    }


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--window-days", type=int, default=30)
    args = parser.parse_args()
    cfg = Config.load()
    import json
    print(json.dumps(compute_cost(cfg, args.window_days), indent=2))


if __name__ == "__main__":
    main()
```
