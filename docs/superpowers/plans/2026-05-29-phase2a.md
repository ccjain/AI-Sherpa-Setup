# Phase 2a Implementation Plan — Full-Transcript Analyzer

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship Phase 2a of AI Sherpa — multi-developer telemetry collection (full session JSONL), indexed storage (SQLite FTS5 + ChromaDB), 14-detector analyzer, Markdown/HTML output, auto-filing of high-confidence findings into the Phase 1 triage queue, plus Anthropic Admin API integration for org-level structured metrics.

**Architecture:** Six components on the existing on-prem Windows server: collector (per laptop, Phase 1.5 design with full-JSONL payload), FastAPI ingest service (writes raw + indexes into SQLite FTS5 + ChromaDB), Admin API puller (daily JSON cache), analyzer (queries all three stores, runs 14 detectors), renderer (Markdown + HTML), auto-filer (`gh issue create` into Phase 1 triage queue). Two-tier retention: 7-day rolling for sessions, forever-with-snapshots for insights.

**Tech Stack:** Python 3.11+, FastAPI + uvicorn (ingest service), pandas (events DataFrame), `sentence-transformers BAAI/bge-small-en-v1.5` + scikit-learn (embeddings), ChromaDB PersistentClient (vector store), SQLite + FTS5 (events + insights), Jinja2 (HTML), `gh` CLI (Issue auto-file), `keyring` (Windows Credential Manager secrets), PowerShell 5.1+ (laptop collector). Phase 0.5 detector code carried forward verbatim.

**Spec:** `docs/superpowers/specs/2026-05-29-phase2a-design.md` (commit `45750f3`).
**Roadmap:** `docs/superpowers/specs/2026-05-28-feedback-program-roadmap.md`.
**Builds on:** Phase 0.5 prototype at `analyzer-prototype/` (merged at `efd9fb4`).

```python
"""Central configuration loaded once at process start.

Every subsystem (ingest, analyzer, pruner, etc.) reads the same Config
instance. Defaults match the design at docs/superpowers/specs/
2026-05-29-phase2a-design.md §11.
"""
from __future__ import annotations
from dataclasses import dataclass, field
from pathlib import Path
import json
import os


@dataclass
class Config:
    root: Path = field(default_factory=lambda: Path(os.environ.get("AI_SHERPA_ROOT", r"D:\ai-sherpa")))

    # Paths (all under root)
    @property
    def sessions_dir(self) -> Path:    return self.root / "sessions"
    @property
    def events_db(self) -> Path:       return self.root / "events.sqlite"
    @property
    def chroma_dir(self) -> Path:      return self.root / "chroma"
    @property
    def insights_db(self) -> Path:     return self.root / "insights.sqlite"
    @property
    def admin_api_cache(self) -> Path: return self.root / "admin_api_cache"
    @property
    def logs_dir(self) -> Path:        return self.root / "logs"
    @property
    def output_dir(self) -> Path:      return self.root / "analyzer-out"

    # Retention
    session_retention_days: int = 7
    admin_api_retention_days: int = 365
    insights_retention_days: int | None = None
    output_retention_days: int = 30

    # Ingest
    ingest_host: str = "0.0.0.0"
    ingest_port: int = 8080
    require_auth: bool = False
    enable_transcripts: bool = True

    # Schedules
    admin_puller_time: str = "02:00"
    analyzer_run_time: str = "02:30"
    pruner_run_time: str = "03:00"

    # Admin API
    admin_api_credential_name: str = "ai-sherpa-admin-api"
    admin_api_backfill_days: int = 8

    # Detector tuning
    detector_overrides: dict = field(default_factory=dict)

    # Auto-file
    auto_file_min_confidence: str = "high"
    auto_file_min_developers: int = 2
    github_repo: str = "ccjain/AI-Sherpa-Setup"
    github_token_credential_name: str = "ai-sherpa-gh-token"

    # Output
    render_markdown: bool = True
    render_html: bool = True

    @classmethod
    def load(cls, path: Path | None = None) -> "Config":
        path = path or Path(os.environ.get("AI_SHERPA_CONFIG", r"D:\ai-sherpa\config.json"))
        if path.exists():
            data = json.loads(path.read_text(encoding="utf-8"))
            return cls(**data)
        return cls()
```

```python
"""DDL for events.sqlite and insights.sqlite. CREATE statements are
idempotent via IF NOT EXISTS."""

EVENTS_DDL = r"""
CREATE TABLE IF NOT EXISTS events (
    event_id        INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id      TEXT NOT NULL,
    session_path    TEXT NOT NULL,
    project_path_hash TEXT NOT NULL,
    developer       TEXT NOT NULL,
    machine_id      TEXT NOT NULL,
    timestamp       TEXT NOT NULL,
    event_type      TEXT NOT NULL,
    text            TEXT,
    tool_name       TEXT,
    tool_args_json  TEXT,
    tool_success    INTEGER,
    command_first_word TEXT,
    file_extension  TEXT,
    skill_name      TEXT,
    slash_command_name TEXT,
    is_first_in_session INTEGER NOT NULL DEFAULT 0
);
CREATE INDEX IF NOT EXISTS idx_events_session   ON events(session_id);
CREATE INDEX IF NOT EXISTS idx_events_developer ON events(developer);
CREATE INDEX IF NOT EXISTS idx_events_type      ON events(event_type);
CREATE INDEX IF NOT EXISTS idx_events_timestamp ON events(timestamp);

CREATE VIRTUAL TABLE IF NOT EXISTS events_fts USING fts5(
    text,
    content='events',
    content_rowid='event_id'
);

CREATE TRIGGER IF NOT EXISTS events_ai AFTER INSERT ON events BEGIN
    INSERT INTO events_fts(rowid, text) VALUES (new.event_id, new.text);
END;
CREATE TRIGGER IF NOT EXISTS events_ad AFTER DELETE ON events BEGIN
    DELETE FROM events_fts WHERE rowid = old.event_id;
END;

CREATE TABLE IF NOT EXISTS sessions (
    session_id      TEXT PRIMARY KEY,
    session_path    TEXT NOT NULL,
    developer       TEXT NOT NULL,
    machine_id      TEXT NOT NULL,
    started_at      TEXT,
    ended_at        TEXT,
    event_count     INTEGER NOT NULL,
    tool_call_count INTEGER NOT NULL,
    nda_skipped     INTEGER NOT NULL DEFAULT 0,
    ingested_at     TEXT NOT NULL
);
CREATE INDEX IF NOT EXISTS idx_sessions_developer ON sessions(developer);
CREATE INDEX IF NOT EXISTS idx_sessions_started   ON sessions(started_at);
"""
```

```python
"""Apply schemas to events.sqlite and insights.sqlite. Idempotent."""
from __future__ import annotations
import sqlite3
from server.config import Config
from server.storage.schema import EVENTS_DDL, INSIGHTS_DDL


def migrate_all(cfg: Config) -> None:
    cfg.root.mkdir(parents=True, exist_ok=True)
    _apply(cfg.events_db, EVENTS_DDL)
    _apply(cfg.insights_db, INSIGHTS_DDL)


def _apply(db_path, ddl: str) -> None:
    with sqlite3.connect(db_path) as conn:
        conn.executescript(ddl)
        conn.commit()
```

```python
"""Persistent ChromaDB client wrapper.

Three collections, all cosine distance:
- first_prompts: first user prompt per session (≤500 chars) — embedded by ingest
- corrections: user prompts matching correction phrases (≤300 chars)
- assistant_responses_sample: sampled assistant responses for future use
"""
from __future__ import annotations
import chromadb
from server.config import Config


COLLECTIONS = ("first_prompts", "corrections", "assistant_responses_sample")


def open_chroma(cfg: Config):
    """Return a ChromaDB PersistentClient with the three collections ensured."""
    cfg.chroma_dir.mkdir(parents=True, exist_ok=True)
    client = chromadb.PersistentClient(path=str(cfg.chroma_dir))
    for name in COLLECTIONS:
        client.get_or_create_collection(
            name=name,
            metadata={"hnsw:space": "cosine"},
        )
    return client
```

```python
def parse_jsonl_string(jsonl_text: str, session_path: str, default_session_id: str) -> list[dict]:
    """Parse a JSONL payload (string) into a list of event dicts.

    Used by the ingest endpoint where we have the raw body in memory rather
    than a file on disk. Reuses _classify_event / _COLUMNS conventions from the
    Phase 0.5 parser carried over above.
    """
    import hashlib, json
    project_path_hash = hashlib.sha256(session_path.encode()).hexdigest()[:16]
    rows: list[dict] = []
    for line in jsonl_text.splitlines():
        if not line.strip():
            continue
        try:
            obj = json.loads(line)
        except json.JSONDecodeError:
            continue
        session_id = obj.get("sessionId") or obj.get("session_id") or default_session_id
        timestamp = obj.get("timestamp")
        for event in _classify_event(obj):
            event.update({
                "session_id": session_id,
                "session_path": session_path,
                "project_path_hash": project_path_hash,
                "timestamp": timestamp,
            })
            for col in _COLUMNS:
                event.setdefault(col, None)
            rows.append(event)
    return rows
```

- [ ] **Step 5: Write `server/ingest/main.py`**

```python
"""FastAPI ingest endpoint.

POST /v1/sessions/{session_id}
  headers: X-Machine-Id, X-Developer
  body:    JSONL (raw bytes)

Behavior:
  1. SHA-256 the body.
  2. If (machine_id, session_id, sha) already in `sessions`, return 200.
  3. Write to D:\ai-sherpa\sessions\<dev>\<machine>\<session>.jsonl
  4. Parse into events; bulk INSERT into events table (FTS5 sync via trigger).
  5. UPSERT into sessions table.
  6. Embed first user prompt into ChromaDB first_prompts collection.
  7. Return 201.

Indexing failures (steps 4-6) are logged but DO NOT fail the upload.
"""
from __future__ import annotations
import datetime
import hashlib
import logging
import sqlite3
from pathlib import Path
from fastapi import FastAPI, Header, HTTPException, Request, Response
from server.config import Config
from server.ingest.parser import parse_jsonl_string
from server.storage.chroma import open_chroma


log = logging.getLogger("ai-sherpa.ingest")


def create_app(cfg: Config) -> FastAPI:
    app = FastAPI(title="AI Sherpa ingest", version="0.1.0")
    cfg.sessions_dir.mkdir(parents=True, exist_ok=True)
    cfg.logs_dir.mkdir(parents=True, exist_ok=True)
    chroma_client = open_chroma(cfg)
    first_prompts_coll = chroma_client.get_collection("first_prompts")

    # Lazy-load embedding model on first request to avoid blocking app startup.
    state = {"embedder": None}

    def _embed(texts: list[str]):
        if state["embedder"] is None:
            from sentence_transformers import SentenceTransformer
            state["embedder"] = SentenceTransformer("BAAI/bge-small-en-v1.5")
        return state["embedder"].encode(texts, normalize_embeddings=True).tolist()

    @app.post("/v1/sessions/{session_id}", status_code=201)
    async def post_session(
        session_id: str,
        request: Request,
        x_machine_id: str = Header(..., alias="X-Machine-Id"),
        x_developer: str = Header(..., alias="X-Developer"),
    ):
        body = await request.body()
        sha = hashlib.sha256(body).hexdigest()

        # Dedup check
        with sqlite3.connect(cfg.events_db) as conn:
            existing = conn.execute(
                "SELECT 1 FROM sessions WHERE session_id=? AND machine_id=? LIMIT 1",
                (session_id, x_machine_id),
            ).fetchone()
        if existing:
            return Response(content='{"status":"already-have-this"}', media_type="application/json", status_code=200)

        # Write raw JSONL
        target_dir = cfg.sessions_dir / x_developer / x_machine_id
        target_dir.mkdir(parents=True, exist_ok=True)
        target_path = target_dir / f"{session_id}.jsonl"
        target_path.write_bytes(body)

        # Parse + index
        text = body.decode("utf-8", errors="replace")
        try:
            event_rows = parse_jsonl_string(text, str(target_path), session_id)
        except Exception as exc:
            log.error("parse failed for %s: %s", session_id, exc)
            event_rows = []

        if event_rows:
            try:
                _insert_events(cfg, event_rows, x_developer, x_machine_id, target_path)
                _embed_first_prompt(event_rows, first_prompts_coll, _embed, x_developer)
            except Exception as exc:
                log.error("indexing failed for %s: %s", session_id, exc)

        # Audit log
        log_path = cfg.logs_dir / "ingest.log"
        log_line = f'{datetime.datetime.utcnow().isoformat()} session={session_id} dev={x_developer} machine={x_machine_id} bytes={len(body)}\n'
        with log_path.open("a", encoding="utf-8") as f:
            f.write(log_line)

        return {"status": "stored", "sha": sha}

    @app.get("/healthz")
    async def healthz():
        return {"status": "ok"}

    return app


def _insert_events(cfg: Config, rows: list[dict], developer: str, machine_id: str, session_path: Path) -> None:
    """Bulk insert events; mark first prompt; UPSERT sessions row."""
    if not rows:
        return
    # Mark first prompt
    for r in rows:
        r["developer"] = developer
        r["machine_id"] = machine_id
    seen_first_prompt = False
    for r in rows:
        if r["event_type"] == "prompt" and not seen_first_prompt:
            r["is_first_in_session"] = 1
            seen_first_prompt = True

    cols = [
        "session_id", "session_path", "project_path_hash", "developer", "machine_id",
        "timestamp", "event_type", "text", "tool_name", "tool_args_json",
        "tool_success", "command_first_word", "file_extension",
        "skill_name", "slash_command_name", "is_first_in_session",
    ]
    placeholders = ", ".join("?" * len(cols))
    sql = f"INSERT INTO events ({', '.join(cols)}) VALUES ({placeholders})"

    with sqlite3.connect(cfg.events_db) as conn:
        conn.executemany(sql, [tuple(r.get(c) for c in cols) for r in rows])

        # UPSERT sessions row
        session_id = rows[0]["session_id"]
        timestamps = [r["timestamp"] for r in rows if r["timestamp"]]
        started_at = min(timestamps) if timestamps else None
        ended_at = max(timestamps) if timestamps else None
        event_count = len(rows)
        tool_call_count = sum(1 for r in rows if r["event_type"] == "tool_call")
        conn.execute(
            "INSERT OR REPLACE INTO sessions "
            "(session_id, session_path, developer, machine_id, started_at, ended_at, "
            "event_count, tool_call_count, nda_skipped, ingested_at) "
            "VALUES (?, ?, ?, ?, ?, ?, ?, ?, 0, ?)",
            (session_id, str(session_path), developer, machine_id,
             started_at, ended_at, event_count, tool_call_count,
             datetime.datetime.utcnow().isoformat()),
        )
        conn.commit()


def _embed_first_prompt(rows: list[dict], collection, embed_fn, developer: str) -> None:
    """Embed the session's first user prompt into the first_prompts ChromaDB collection."""
    first_prompts = [r for r in rows if r.get("is_first_in_session")]
    if not first_prompts:
        return
    fp = first_prompts[0]
    text = (fp.get("text") or "")[:500]
    if not text.strip():
        return
    embedding = embed_fn([text])[0]
    session_id = fp["session_id"]
    collection.add(
        ids=[session_id],
        embeddings=[embedding],
        documents=[text],
        metadatas=[{
            "session_id": session_id,
            "developer": developer,
            "timestamp": fp.get("timestamp") or "",
        }],
    )
```

```python
from pathlib import Path


def safe_file_path_from_args(j) -> str | None:
    """Pull file_path from a tool_args_json string."""
    if not isinstance(j, str):
        return None
    try:
        return json.loads(j).get("file_path")
    except (json.JSONDecodeError, AttributeError):

def lookup_session_paths(session_ids: list[str], cfg) -> list[str]:
    """Query sessions table for session_paths. Returns '<session pruned>' for missing."""
    import sqlite3
    if not session_ids:
        return []
    placeholders = ", ".join("?" * len(session_ids))
    with sqlite3.connect(cfg.events_db) as conn:
        rows = dict(conn.execute(
            f"SELECT session_id, session_path FROM sessions WHERE session_id IN ({placeholders})",
            session_ids,
        ).fetchall())
    return [rows.get(sid, "<session pruned>") for sid in session_ids]
```

```python
# ── Scenario 16 — Edits without prior Read ─────────────────────────────

def detect_edits_without_prior_read(events: pd.DataFrame, embeddings_fn=None, **ctx) -> Iterable[Finding]:
    if events.empty:
        return []
    tool_calls = events[
        (events.event_type == "tool_call") &
        events.tool_name.isin(["Read", "Edit", "Write"])
    ].copy()
    if tool_calls.empty:
        return []
    tool_calls["file_path"] = tool_calls.tool_args_json.apply(safe_file_path_from_args)
    tool_calls = tool_calls[tool_calls.file_path.notna()].copy()
    if tool_calls.empty:
        return []

    edits_without_read = 0
    total_edits = 0
    sample_session_ids: list[str] = []
    for (sid, fp), group in tool_calls.groupby(["session_id", "file_path"]):
        group = group.sort_values("timestamp")
        types = group.tool_name.tolist()
        first_read_idx = next((i for i, t in enumerate(types) if t == "Read"), None)
        first_edit_idx = next((i for i, t in enumerate(types) if t in ("Edit", "Write")), None)
        if first_edit_idx is None:
            continue
        total_edits += 1
        if first_read_idx is None or first_read_idx > first_edit_idx:
            edits_without_read += 1
            if len(sample_session_ids) < 10:
                sample_session_ids.append(sid)

    if total_edits < 10:
        return []
    ratio = edits_without_read / total_edits
    if ratio < 0.30:
        return []

    return [Finding(
        scenario_id="scenario-16",
        title=f"{ratio:.0%} of file edits had no prior Read in same session",
        bucket="add-rule",
        domain=None,
        severity="high" if ratio >= 0.50 else "normal",
        confidence="high" if total_edits >= 50 else "medium",
        evidence_md=(

            f"**{edits_without_read} ({ratio:.0%})** were edited without first being read. "
            f"Anthropic's published research on quality regression found this ratio at 43.8%; "
            f"yours is **{ratio:.0%}**.\n\n"
            f"This is the strongest single correlate of Claude making changes to code it "
            f"doesn't understand — and the strongest predictor of post-edit corrections.\n\n"
            f"**Suggested change:** add a rule to `core/CLAUDE.md` requiring `Read` before "
            f"any `Edit`/`Write` to a file Claude hasn't seen this session."
        ),
        sample_session_paths=[],   # filled in by analyzer with helpers.lookup_session_paths
```

```python
"""Embeddings_fn that prefers ChromaDB-indexed embeddings, falling back to
on-demand encoding for texts that aren't already indexed.

Used by all embedding-based detectors (scenarios 1, 2, 5, 9).
"""
from __future__ import annotations
from typing import Callable
import numpy as np
from server.config import Config
from server.storage.chroma import open_chroma


def make_embeddings_fn(cfg: Config) -> Callable[[list[str]], np.ndarray]:
    """Return a function that maps texts → 2-D numpy array.

    For each text:
    - Look up via document match in `first_prompts` collection
    - If found, return the indexed embedding
    - Else, fall back to live encoding via sentence-transformers

    Live-encoded results are NOT auto-inserted into chroma — the ingest service
    is the only authoritative inserter.
    """
    client = open_chroma(cfg)
    first_prompts = client.get_collection("first_prompts")

    state = {"encoder": None}

    def _ensure_encoder():
        if state["encoder"] is None:
            from sentence_transformers import SentenceTransformer
            state["encoder"] = SentenceTransformer("BAAI/bge-small-en-v1.5")
        return state["encoder"]

    def fn(texts: list[str]) -> np.ndarray:
        if not texts:
            return np.zeros((0, 384))
        # Query chroma for documents matching each text. ChromaDB doesn't do
        # exact document lookup natively, so we fall back to live encoding
        # for any text that doesn't yield a match. For the prototype we encode
        # all texts live; the chroma collection is used for persistence across
        # runs via the ingest service's writes, not as a query cache here.
        encoder = _ensure_encoder()
        return encoder.encode(texts, normalize_embeddings=True)

    return fn
```

(Note: the v1 keeps it simple — always live-encode at query time. ChromaDB's persistence still pays off because the **ingest** path embeds once per session, so the analyzer's query-time work is bounded by the *number of texts the detectors look at*, not the *number of sessions in the corpus*. Future optimization: pre-build a text→embedding cache from chroma at analyzer startup.)

```python
from __future__ import annotations
import json
from datetime import date
from pathlib import Path
import pandas as pd
from server.config import Config


def load_admin_metrics(cfg: Config, since: date | None = None) -> pd.DataFrame:
    cache = cfg.admin_api_cache
    if not cache.exists():
        return pd.DataFrame()
    rows = []
    for p in sorted(cache.glob("*.json")):
        try:
            d = date.fromisoformat(p.stem)
        except ValueError:
            continue
        if since and d < since:
            continue
        try:
            data = json.loads(p.read_text(encoding="utf-8"))
        except json.JSONDecodeError:
            continue
        for entry in data.get("daily_user_metrics", []):
            rows.append({"date": d, **entry})
    return pd.DataFrame(rows)
```

```python
"""Daily Admin API pull. Run via Windows Task Scheduler at admin_puller_time.

Fetches the last `admin_api_backfill_days` days each run, idempotently
overwrites cached files. Skipping a day self-heals next run.
"""
from __future__ import annotations
import datetime
import json
import sys
import time
from server.config import Config

ENDPOINT = "https://api.anthropic.com/v1/organizations/usage"


def get_admin_api_key(cfg: Config) -> str:
    import keyring
    key = keyring.get_password(cfg.admin_api_credential_name, "default")
    if not key:
        raise RuntimeError(
            f"Admin API key not found in Windows Credential Manager under "
            f"'{cfg.admin_api_credential_name}' / 'default'. "
            f"Set it with: py -c \"import keyring; keyring.set_password('{cfg.admin_api_credential_name}', 'default', 'sk-...')\""
        )
    return key


def pull_for_date(cfg: Config, d: datetime.date, api_key: str) -> dict:
    import requests
    resp = requests.get(
        ENDPOINT,
        headers={"x-api-key": api_key, "anthropic-version": "2023-06-01"},
        params={"date": d.isoformat()},
        timeout=60,
    )
    if resp.status_code in (429, 500, 502, 503, 504):
        time.sleep(60)
        resp = requests.get(
            ENDPOINT,
            headers={"x-api-key": api_key, "anthropic-version": "2023-06-01"},
            params={"date": d.isoformat()},
            timeout=60,
        )
    resp.raise_for_status()
    return resp.json()


def main():
    cfg = Config.load()
    cfg.admin_api_cache.mkdir(parents=True, exist_ok=True)
    api_key = get_admin_api_key(cfg)
    today = datetime.date.today()
    pulled = 0
    for offset in range(0, cfg.admin_api_backfill_days):
        d = today - datetime.timedelta(days=offset)
        out_path = cfg.admin_api_cache / f"{d.isoformat()}.json"
        if out_path.exists() and offset > 0:
            continue
        try:
            data = pull_for_date(cfg, d, api_key)
        except Exception as exc:
            print(f"[admin_api] failed for {d}: {exc}", file=sys.stderr)
            continue
        out_path.write_text(json.dumps(data, indent=2), encoding="utf-8")
        pulled += 1
        print(f"[admin_api] cached {d}")
    print(f"[admin_api] done; {pulled} day(s) updated")


if __name__ == "__main__":
    main()
```

```python
"""Helpers for persisting findings and detecting recurrences."""
from __future__ import annotations
import datetime
import hashlib
import json
import sqlite3
import numpy as np
from typing import Any
from server.config import Config
from server.analyzer.detectors.base import Finding


def start_run(cfg: Config, sessions_seen: int) -> int:
    with sqlite3.connect(cfg.insights_db) as conn:
        cur = conn.execute(
            "INSERT INTO runs (started_at, sessions_seen) VALUES (?, ?)",
            (datetime.datetime.utcnow().isoformat(), sessions_seen),
        )
        conn.commit()
        return cur.lastrowid


def finish_run(cfg: Config, run_id: int, findings_total: int, error_log_json: str | None) -> None:
    with sqlite3.connect(cfg.insights_db) as conn:
        conn.execute(
            "UPDATE runs SET ended_at=?, findings_total=?, error_log_json=? WHERE run_id=?",
            (datetime.datetime.utcnow().isoformat(), findings_total, error_log_json, run_id),
        )
        conn.commit()


def _fingerprint(scenario_id: str, title: str, contributing_session_ids: list[str]) -> str:
    h = hashlib.sha256()
    h.update(scenario_id.encode())
    h.update(b"|")
    h.update(title.strip().lower().encode())
    h.update(b"|")
    h.update(",".join(sorted(contributing_session_ids)).encode())
    return h.hexdigest()


def insert_finding(
    cfg: Config, run_id: int, finding: Finding,
    cluster_centroid: np.ndarray | None,
    sample_snippets: list[dict],
    contributing_session_ids: list[str],
    contributing_developer_count: int,
    contributing_occurrence_count: int,
) -> int:
    fingerprint = _fingerprint(finding.scenario_id, finding.title, contributing_session_ids)
    centroid_blob = cluster_centroid.astype(np.float32).tobytes() if cluster_centroid is not None else None

    with sqlite3.connect(cfg.insights_db) as conn:
        cur = conn.execute(
            "INSERT INTO findings (run_id, scenario_id, title, bucket, domain, severity, "
            "confidence, evidence_md, fingerprint, cluster_centroid_embedding, "
            "sample_snippets_json, contributing_session_ids_json, "
            "contributing_developer_count, contributing_occurrence_count) "
            "VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)",
            (
                run_id, finding.scenario_id, finding.title, finding.bucket,
                finding.domain, finding.severity, finding.confidence,
                finding.evidence_md, fingerprint, centroid_blob,
                json.dumps(sample_snippets), json.dumps(contributing_session_ids),
                contributing_developer_count, contributing_occurrence_count,
            ),
        )
        conn.commit()
        return cur.lastrowid


def find_matching_past_finding(
    cfg: Config, scenario_id: str, centroid: np.ndarray, threshold: float = 0.92,
) -> dict | None:
    """Return the first past finding of `scenario_id` whose centroid has
    cosine similarity > threshold, or None."""
    with sqlite3.connect(cfg.insights_db) as conn:
        rows = conn.execute(
            "SELECT finding_id, cluster_centroid_embedding, contributing_occurrence_count "
            "FROM findings WHERE scenario_id=? AND cluster_centroid_embedding IS NOT NULL "
            "ORDER BY finding_id DESC LIMIT 200",
            (scenario_id,),
        ).fetchall()
    q = centroid.astype(np.float32)
    q_norm = np.linalg.norm(q)
    if q_norm == 0:
        return None
    for fid, blob, occ in rows:
        past = np.frombuffer(blob, dtype=np.float32)
        if past.shape != q.shape:
            continue
        sim = float(np.dot(q, past) / (q_norm * np.linalg.norm(past) + 1e-9))
        if sim >= threshold:
            return {"finding_id": fid, "similarity": sim, "occurrence_count": occ}
    return None


def increment_recurrence(cfg: Config, finding_id: int, run_id: int) -> None:
    with sqlite3.connect(cfg.insights_db) as conn:
        # Update occurrence count
        conn.execute(
            "UPDATE findings SET contributing_occurrence_count = contributing_occurrence_count + 1 "
            "WHERE finding_id=?",
            (finding_id,),
        )
        # Insert recurrence row
        new_count = conn.execute(
            "SELECT contributing_occurrence_count FROM findings WHERE finding_id=?",
            (finding_id,),
        ).fetchone()[0]
        conn.execute(
            "INSERT OR REPLACE INTO finding_recurrences "
            "(finding_id, run_id, occurrence_count_at_run) VALUES (?, ?, ?)",
            (finding_id, run_id, new_count),
        )
        conn.commit()
```

```python
"""Analyzer pipeline orchestrator.

Reads from events.sqlite + chroma + admin_api_cache, runs the 14 detectors,
writes findings to insights.sqlite (with recurrence check), renders Markdown
+ HTML, and optionally auto-files high-confidence findings as Issues.
"""
from __future__ import annotations
import datetime
import inspect
import json
import logging
import sqlite3
import sys
from pathlib import Path
import numpy as np
import pandas as pd
from server.config import Config
from server.storage.migrate import migrate_all
from server.analyzer.detectors import ALL_DETECTORS
from server.analyzer.detectors._helpers import lookup_session_paths
from server.analyzer.embeddings import make_embeddings_fn
from server.analyzer.insights import (
    start_run, finish_run, insert_finding, find_matching_past_finding,
    increment_recurrence,
)
from server.admin_api.normalizer import load_admin_metrics


log = logging.getLogger("ai-sherpa.analyzer")


def _events_from_window(cfg: Config, days: int) -> pd.DataFrame:
    cutoff = (datetime.datetime.utcnow() - datetime.timedelta(days=days)).isoformat()
    with sqlite3.connect(cfg.events_db) as conn:
        df = pd.read_sql_query(
            "SELECT * FROM events WHERE timestamp >= ?", conn, params=(cutoff,),
        )
    if not df.empty:
        df["timestamp"] = pd.to_datetime(df["timestamp"], utc=True, errors="coerce")
        df["is_first_in_session"] = df["is_first_in_session"].astype(bool)
    return df


def _context_for_detectors(cfg: Config, events: pd.DataFrame) -> dict:
    return {
        "configured_domain": None,        # Phase 2a doesn't infer a single domain — set per finding
        "installed_skills": [],            # not used in v1
        "current_version": None,           # not used in v1
        "detector_overrides": cfg.detector_overrides,
        "admin_metrics": load_admin_metrics(cfg),
    }


def _call_detector(d, events, embeddings_fn, context):
    sig = inspect.signature(d)
    extra = {k: v for k, v in context.items() if k in sig.parameters}
    return list(d(events, embeddings_fn, **extra))


def _centroid_for_finding(finding, embeddings_fn) -> np.ndarray | None:
    """Compute a centroid embedding so we can detect recurrence later.

    For embedding-based detectors that already produce a centroid in the
    finding's metadata, this could be passed through. For v1 we embed the
    finding's title + first 200 chars of evidence_md as a proxy.
    """
    text = (finding.title or "") + " " + (finding.evidence_md or "")[:200]
    if not text.strip():
        return None
    embs = embeddings_fn([text])
    if embs.size == 0:
        return None
    return embs[0]


def run_pipeline(cfg: Config, auto_file: bool = True) -> dict:
    """Run one analyzer pass. Returns a dict with run_id and findings_total."""
    migrate_all(cfg)
    events = _events_from_window(cfg, cfg.session_retention_days)
    sessions_seen = events.session_id.nunique() if not events.empty else 0
    run_id = start_run(cfg, sessions_seen=sessions_seen)

    embeddings_fn = make_embeddings_fn(cfg)
    context = _context_for_detectors(cfg, events)

    errors: list[dict] = []
    new_findings: list[tuple[int, "Finding"]] = []

    for d in ALL_DETECTORS:
        try:
            produced = _call_detector(d, events, embeddings_fn, context)
        except Exception as exc:
            log.exception("detector %s failed", getattr(d, "id", d.__name__))
            errors.append({"detector": getattr(d, "id", d.__name__), "error": str(exc)})
            continue

        for finding in produced:
            centroid = _centroid_for_finding(finding, embeddings_fn)
            past = None
            if centroid is not None:
                past = find_matching_past_finding(cfg, finding.scenario_id, centroid)

            if past is not None:
                increment_recurrence(cfg, past["finding_id"], run_id)
                continue

            # Build snapshot data
            sample_snippets = [{"path": p} for p in finding.sample_session_paths[:5]]
            contributing_session_ids = []   # detectors may surface these in evidence; v1 uses sample paths
            contributing_developer_count = 1
            contributing_occurrence_count = 1

            fid = insert_finding(
                cfg, run_id=run_id, finding=finding,
                cluster_centroid=centroid,
                sample_snippets=sample_snippets,
                contributing_session_ids=contributing_session_ids,
                contributing_developer_count=contributing_developer_count,
                contributing_occurrence_count=contributing_occurrence_count,
            )
            new_findings.append((fid, finding))

    err_json = json.dumps(errors) if errors else None
    finish_run(cfg, run_id, findings_total=len(new_findings), error_log_json=err_json)

    # Render + auto-file (separate concerns; both query insights.sqlite)
    if cfg.render_markdown or cfg.render_html:
        from server.analyzer.render import render_run
        render_run(cfg, run_id)
    if auto_file:
        from server.analyzer.auto_filer import auto_file_run
        auto_file_run(cfg, run_id)

    return {"run_id": run_id, "findings_total": len(new_findings), "errors": errors}


def main():
    cfg = Config.load()
    result = run_pipeline(cfg, auto_file=True)
    print(json.dumps(result, indent=2))


if __name__ == "__main__":
    main()
```

```python
"""Render findings from insights.sqlite into Markdown + HTML."""
from __future__ import annotations
import datetime
import json
import re
import sqlite3
from pathlib import Path
from jinja2 import Environment, FileSystemLoader, select_autoescape
from server.config import Config


_SEVERITY_ORDER = {"critical": 0, "high": 1, "normal": 2, "low": 3}


def _slug(s: str, max_len: int = 60) -> str:
    s = re.sub(r"[^a-zA-Z0-9]+", "-", s.lower()).strip("-")
    return s[:max_len].rstrip("-")


def _render_one(row: dict) -> str:
    snippets = json.loads(row.get("sample_snippets_json") or "[]")
    sample_block = "\n".join(f"- `{s.get('path', s)}`" for s in snippets) or "_(no sample sessions captured)_"
    return (
        f"# [{row['bucket']}] {row['title']}\n\n"
        f"**Scenario:** {row['scenario_id']} (see roadmap §3)\n"
        f"**Domain:** {row.get('domain') or 'any'}\n"
        f"**Severity:** {row['severity']}\n"
        f"**Confidence:** {row['confidence']}\n\n"
        f"## Suggested change\n\n"
        f"{row['evidence_md']}\n\n"
        f"## Evidence summary\n\n"
        f"- Contributing sessions: {row.get('contributing_occurrence_count', 1)}\n"
        f"- Contributing developers: {row.get('contributing_developer_count', 1)}\n"
        f"- Fingerprint: `{(row.get('fingerprint') or '')[:12]}`\n\n"
        f"## Sample sessions\n\n"
        f"{sample_block}\n"
    )


def render_run(cfg: Config, run_id: int) -> Path:
    """Render every finding for a run into Markdown + HTML files. Returns
    the output directory."""
    today = datetime.date.today().isoformat()
    out_dir = cfg.output_dir / today
    issues_dir = out_dir / "issues"
    issues_dir.mkdir(parents=True, exist_ok=True)

    with sqlite3.connect(cfg.insights_db) as conn:
        conn.row_factory = sqlite3.Row
        rows = list(conn.execute("SELECT * FROM findings WHERE run_id=?", (run_id,)))

    findings = [dict(r) for r in rows]
    findings.sort(key=lambda f: (_SEVERITY_ORDER.get(f["severity"], 99), f["scenario_id"], f["title"]))

    written_paths = []
    for i, f in enumerate(findings, start=1):
        filename = f"{i:03d}-{_slug(f['title'])}.md"
        path = issues_dir / filename
        path.write_text(_render_one(f), encoding="utf-8")
        written_paths.append(path)
        # Attach filename for HTML template
        f["_filename"] = filename

    # Render HTML summary
    template_dir = Path(__file__).parent / "templates"
    env = Environment(
        loader=FileSystemLoader(str(template_dir)),
        autoescape=select_autoescape(["html", "xml"]),
    )
    template = env.get_template("summary.html.j2")

    # Map session_count for the meta block
    with sqlite3.connect(cfg.insights_db) as conn:
        sessions_seen = conn.execute(
            "SELECT sessions_seen FROM runs WHERE run_id=?", (run_id,)
        ).fetchone()
    session_count = sessions_seen[0] if sessions_seen else 0

    html = template.render(
        findings=findings,
        generated_at=datetime.datetime.utcnow().isoformat(timespec="seconds"),
        input_path=str(cfg.sessions_dir),
        session_count=session_count,
    )
    (out_dir / "summary.html").write_text(html, encoding="utf-8")

    # Update the `latest` junction
    latest = cfg.output_dir / "latest"
    if latest.exists() or latest.is_symlink():
        try:
            latest.unlink()
        except OSError:
            pass
    try:
        latest.symlink_to(out_dir, target_is_directory=True)
    except OSError:
        pass   # Windows non-admin may not allow symlinks; that's fine

    return out_dir
```

```python
"""Auto-file qualifying findings as GitHub Issues into AI Sherpa repo.

Qualification:
- confidence >= cfg.auto_file_min_confidence
- contributing_developer_count >= cfg.auto_file_min_developers
- not already filed (issue_url IS NULL)
"""
from __future__ import annotations
import datetime
import json
import os
import sqlite3
import subprocess
import tempfile
from pathlib import Path
from server.config import Config


_CONFIDENCE_ORDER = {"high": 0, "medium": 1, "low": 2}


def _get_github_token(cfg: Config) -> str:
    import keyring
    token = keyring.get_password(cfg.github_token_credential_name, "default")
    if not token:
        raise RuntimeError(
            f"GitHub PAT not found in Windows Credential Manager under "
            f"'{cfg.github_token_credential_name}' / 'default'."
        )
    return token


def _gh_issue_create(cfg: Config, title: str, body: str, labels: list[str]) -> str:
    """Run `gh issue create` and return the resulting URL."""
    with tempfile.NamedTemporaryFile("w", encoding="utf-8", suffix=".md", delete=False) as f:
        f.write(body)
        body_path = f.name
    try:
        cmd = [
            "gh", "issue", "create",
            "--repo", cfg.github_repo,
            "--title", title,
            "--body-file", body_path,
        ]
        for label in labels:
            cmd.extend(["--label", label])
        result = subprocess.run(
            cmd, capture_output=True, text=True, check=True,
            env={**os.environ, "GH_TOKEN": _get_github_token(cfg)},
        )
        return result.stdout.strip()
    finally:
        try:
            os.unlink(body_path)
        except OSError:
            pass


def _build_body(row: dict) -> str:
    snippets = json.loads(row.get("sample_snippets_json") or "[]")
    sample_block = "\n".join(f"- `{s.get('path', s)}`" for s in snippets) or "_(no sample sessions captured)_"
    return (
        f"## Suggested change\n\n{row['evidence_md']}\n\n"
        f"## Evidence summary\n\n"
        f"- Contributing sessions: {row['contributing_occurrence_count']}\n"
        f"- Contributing developers: {row['contributing_developer_count']}\n"
        f"- Fingerprint: `{(row['fingerprint'] or '')[:12]}`\n\n"
        f"## Sample sessions\n\n"
        f"(sessions older than 7 days are no longer on the server but were captured at find-time)\n\n"
        f"{sample_block}\n"
    )


def auto_file_run(cfg: Config, run_id: int) -> None:
    """File qualifying findings from a given run."""
    min_conf_idx = _CONFIDENCE_ORDER.get(cfg.auto_file_min_confidence, 0)

    with sqlite3.connect(cfg.insights_db) as conn:
        conn.row_factory = sqlite3.Row
        rows = list(conn.execute(
            "SELECT * FROM findings WHERE run_id=? AND issue_url IS NULL", (run_id,)
        ))

    for r in rows:
        if _CONFIDENCE_ORDER.get(r["confidence"], 99) > min_conf_idx:
            continue
        if r["contributing_developer_count"] < cfg.auto_file_min_developers:
            continue
        title = f"[{r['bucket']}] {r['title']}"
        labels = [
            "feedback", "source:telemetry",
            f"type/{r['bucket']}",
            f"severity/{r['severity']}",
            f"confidence/{r['confidence']}",
            "status/needs-review",
        ]
        if r.get("domain"):
            labels.append(f"domain/{r['domain']}")

        try:
            url = _gh_issue_create(cfg, title, _build_body(dict(r)), labels)
        except Exception as exc:
            print(f"[auto_filer] failed to file {r['finding_id']}: {exc}")
            continue

        with sqlite3.connect(cfg.insights_db) as conn2:
            conn2.execute(
                "UPDATE findings SET issue_url=?, issue_filed_at=? WHERE finding_id=?",
                (url, datetime.datetime.utcnow().isoformat(), r["finding_id"]),
            )
            conn2.commit()
        print(f"[auto_filer] filed {r['finding_id']} → {url}")
```

```python
"""Daily pruner — runs at pruner_run_time. Three jobs in sequence:
1. Session pruning (events table + raw JSONL files)
2. Admin cache pruning
3. Output folder pruning
4. ChromaDB metadata-filtered deletion of old embeddings
"""
from __future__ import annotations
import datetime
import sqlite3
from pathlib import Path
from server.config import Config
from server.storage.chroma import open_chroma


def prune(cfg: Config) -> dict:
    cutoff = (datetime.datetime.utcnow() - datetime.timedelta(days=cfg.session_retention_days)).isoformat()
    counts = {"sessions": 0, "events": 0, "admin_files": 0, "output_dirs": 0, "chroma": 0}

    # 1. Sessions + events
    with sqlite3.connect(cfg.events_db) as conn:
        # Collect session_ids to delete
        old_session_ids = [
            r[0] for r in conn.execute(
                "SELECT session_id FROM sessions WHERE ended_at < ?", (cutoff,)
            )
        ]
        # Delete raw files
        for sid in old_session_ids:
            row = conn.execute(
                "SELECT session_path FROM sessions WHERE session_id=?", (sid,)
            ).fetchone()
            if row:
                p = Path(row[0])
                if p.exists():
                    try:
                        p.unlink()
                    except OSError:
                        pass
        # Delete events (FTS5 sync via trigger)
        cur = conn.execute("DELETE FROM events WHERE timestamp < ?", (cutoff,))
        counts["events"] = cur.rowcount
        # Delete sessions
        cur2 = conn.execute("DELETE FROM sessions WHERE ended_at < ?", (cutoff,))
        counts["sessions"] = cur2.rowcount
        conn.commit()

    # 2. Admin cache
    cutoff_admin = datetime.date.today() - datetime.timedelta(days=cfg.admin_api_retention_days)
    if cfg.admin_api_cache.exists():
        for p in cfg.admin_api_cache.glob("*.json"):
            try:
                d = datetime.date.fromisoformat(p.stem)
            except ValueError:
                continue
            if d < cutoff_admin:
                try:
                    p.unlink()
                    counts["admin_files"] += 1
                except OSError:
                    pass

    # 3. Output folders
    if cfg.output_dir.exists():
        cutoff_out = datetime.date.today() - datetime.timedelta(days=cfg.output_retention_days)
        for sub in cfg.output_dir.iterdir():
            if not sub.is_dir() or sub.name == "latest":
                continue
            try:
                d = datetime.date.fromisoformat(sub.name)
            except ValueError:
                continue
            if d < cutoff_out:
                import shutil
                shutil.rmtree(sub, ignore_errors=True)
                counts["output_dirs"] += 1

    # 4. ChromaDB metadata-filtered deletion
    try:
        client = open_chroma(cfg)
        for coll_name in ("first_prompts", "corrections", "assistant_responses_sample"):
            coll = client.get_collection(coll_name)
            try:
                coll.delete(where={"timestamp": {"$lt": cutoff}})
                counts["chroma"] += 1
            except Exception:
                pass
    except Exception:
        pass

    return counts


def main():
    cfg = Config.load()
    result = prune(cfg)
    import json
    print(json.dumps(result, indent=2))


if __name__ == "__main__":
    main()
```

```powershell
# AI Sherpa client uploader.
# Modes:
#   -Mode Hook   : called by Claude Code SessionEnd hook; uploads the ended session
#   -Mode Sweep  : hourly Task Scheduler entry; scans for new + retries pending

param(
    [Parameter(Mandatory=$true)][ValidateSet("Hook","Sweep")]
    [string]$Mode,

    # Server URL (set during setup.bat --enable-telemetry; falls back to default)
    [string]$ServerUrl = $env:AI_SHERPA_SERVER_URL,

    # Override machine id (testing only)
    [string]$MachineIdOverride = ""
)

$ErrorActionPreference = "Stop"
$ProgressPreference = "SilentlyContinue"

if (-not $ServerUrl) { $ServerUrl = "http://ai-sherpa-srv.local:8080" }

$base = Join-Path $env:USERPROFILE ".ai-sherpa"
$dbPath = Join-Path $base "uploads.db"
$machineIdPath = Join-Path $base "machine-id"
$projectsDir = Join-Path $env:USERPROFILE ".claude\projects"

if (-not (Test-Path $base)) { New-Item -ItemType Directory -Path $base | Out-Null }

# Machine ID
if ($MachineIdOverride) {
    $machineId = $MachineIdOverride
} elseif (-not (Test-Path $machineIdPath)) {
    $machineId = [guid]::NewGuid().ToString()
    Set-Content -Path $machineIdPath -Value $machineId
} else {
    $machineId = (Get-Content $machineIdPath).Trim()
}

$developer = $env:USERNAME

# Initialize sqlite uploads.db if missing (PowerShell uses System.Data.SQLite via pip-installed Python is heavy;
# we use a JSON manifest instead for simplicity — sufficient at this scale)
$manifestPath = Join-Path $base "uploads.json"
if (-not (Test-Path $manifestPath)) {
    "{}" | Out-File -FilePath $manifestPath -Encoding utf8
}
$manifest = Get-Content $manifestPath -Raw | ConvertFrom-Json -AsHashtable

function Test-NdaGate {
    param([string]$ProjectDir)
    $cur = $ProjectDir
    while ($cur -and (Test-Path $cur)) {
        foreach ($marker in @("NDA.md", "CONFIDENTIAL.md", ".ai-sherpa-noupload")) {
            if (Test-Path (Join-Path $cur $marker)) { return $false }
        }
        $parent = Split-Path -Parent $cur
        if ($parent -eq $cur) { break }
        $cur = $parent
    }
    return $true
}

function Upload-Session {
    param([System.IO.FileInfo]$JsonlFile)

    $sessionId = $JsonlFile.BaseName
    $sha = (Get-FileHash $JsonlFile.FullName -Algorithm SHA256).Hash

    # Already-uploaded check
    if ($manifest.ContainsKey($sessionId)) {
        if ($manifest[$sessionId].sha -eq $sha -and $manifest[$sessionId].status -eq "uploaded") {
            return $true
        }
    }

    # NDA gate (look at parent project dir, which we infer from session_path metadata is complex;
    # fall back to the session file's parent directory)
    if (-not (Test-NdaGate $JsonlFile.Directory.FullName)) {
        $manifest[$sessionId] = @{ sha = $sha; status = "skipped_nda" }
        return $true
    }

    try {
        $body = [System.IO.File]::ReadAllBytes($JsonlFile.FullName)
        $uri = "$ServerUrl/v1/sessions/$sessionId"
        $headers = @{
            "X-Machine-Id" = $machineId
            "X-Developer" = $developer
            "Content-Type" = "application/octet-stream"
        }
        $resp = Invoke-RestMethod -Uri $uri -Method POST -Body $body -Headers $headers -TimeoutSec 30
        $manifest[$sessionId] = @{
            sha = $sha
            status = "uploaded"
            uploaded_at = (Get-Date -Format o)
        }
        return $true
    } catch {
        $manifest[$sessionId] = @{ sha = $sha; status = "pending"; last_error = $_.Exception.Message }
        return $false
    }
}

if ($Mode -eq "Hook") {
    # Hook fires with the session's path in an env var or arg; fall back to "most recent jsonl"
    if (Test-Path $projectsDir) {
        $latest = Get-ChildItem -Path $projectsDir -Filter *.jsonl -Recurse |
                  Sort-Object LastWriteTime -Descending |
                  Select-Object -First 1
        if ($latest) {
            Upload-Session -JsonlFile $latest | Out-Null
        }
    }
} elseif ($Mode -eq "Sweep") {
    if (Test-Path $projectsDir) {
        Get-ChildItem -Path $projectsDir -Filter *.jsonl -Recurse | ForEach-Object {
            Upload-Session -JsonlFile $_ | Out-Null
        }
    }
```

## Task 18: Index-rebuild tool

`tools/rebuild-indexes.py` deletes `events.sqlite` + `chroma/` and recreates them from raw JSONL. Recovery mechanism.

**Files:**
- Create: `tools/rebuild-indexes.py`

- [ ] **Step 1: Write the script**

```python
#!/usr/bin/env python
"""Rebuild events.sqlite and chroma/ from raw JSONL files in sessions/.

Use after schema changes, suspected corruption, or major version bumps.
"""
from __future__ import annotations
import shutil
import sys
from pathlib import Path
from server.config import Config
from server.storage.migrate import migrate_all
from server.storage.chroma import open_chroma
from server.ingest.main import _insert_events, _embed_first_prompt
from server.ingest.parser import parse_jsonl_string


def rebuild(cfg: Config) -> dict:
    # Wipe indexes
    if cfg.events_db.exists():
        cfg.events_db.unlink()
    if cfg.chroma_dir.exists():
        shutil.rmtree(cfg.chroma_dir)

    # Recreate schemas + chroma collections
    migrate_all(cfg)
    client = open_chroma(cfg)
    first_prompts_coll = client.get_collection("first_prompts")

    # Load embedding model
    from sentence_transformers import SentenceTransformer
    encoder = SentenceTransformer("BAAI/bge-small-en-v1.5")
    def embed_fn(texts):
        return encoder.encode(texts, normalize_embeddings=True).tolist()

    counts = {"sessions": 0, "events": 0}

    # Walk sessions/
    for jsonl_path in sorted(cfg.sessions_dir.rglob("*.jsonl")):
        # Path: sessions/<dev>/<machine>/<session>.jsonl
        try:
            parts = jsonl_path.relative_to(cfg.sessions_dir).parts
            developer = parts[0]
            machine_id = parts[1]
            session_id = jsonl_path.stem
        except (ValueError, IndexError):
            continue
        text = jsonl_path.read_text(encoding="utf-8", errors="replace")
        rows = parse_jsonl_string(text, str(jsonl_path), session_id)
        if not rows:
            continue
        try:
            _insert_events(cfg, rows, developer, machine_id, jsonl_path)
            _embed_first_prompt(rows, first_prompts_coll, embed_fn, developer)
        except Exception as exc:
            print(f"  failed to ingest {session_id}: {exc}", file=sys.stderr)
            continue
        counts["sessions"] += 1
        counts["events"] += len(rows)
        if counts["sessions"] % 100 == 0:
            print(f"  {counts['sessions']} sessions / {counts['events']} events...")

    return counts


def main():
    cfg = Config.load()
    print(f"Rebuilding indexes for {cfg.root}...")
    counts = rebuild(cfg)
    print(f"Done. {counts['sessions']} sessions, {counts['events']} events.")


if __name__ == "__main__":
    main()
```
